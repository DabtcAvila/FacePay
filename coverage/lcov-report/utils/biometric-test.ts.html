import { WebAuthnService } from '@/services/webauthn'
 
export async function testBiometricCapabilities() {
  try {
    console.log('üîç Testing biometric capabilities...')
    
    // Check biometric support
    const capabilities = await WebAuthnService.checkBrowserCapabilities()
    
    console.log('üìä Biometric Capabilities Report:')
    console.log('‚ïê'.repeat(50))
    console.log(`‚úì WebAuthn Supported: ${capabilities.isSupported}`)
    console.log(`‚úì Platform Authenticator Available: ${capabilities.isPlatformAuthenticatorAvailable}`)
    console.log(`‚úì User Verification Supported: ${capabilities.isPlatformAuthenticatorAvailable}`)
    console.log(`‚úì Biometric Types: ${capabilities.biometricTypes.join(', ')}`)
    
    console.log('\nüîß Specific Biometric Support:')
    console.log(`  ‚Ä¢ Face ID (iOS): ${capabilities.biometricAvailability.faceID}`)
    console.log(`  ‚Ä¢ Touch ID (iOS/macOS): ${capabilities.biometricAvailability.touchID}`)
    console.log(`  ‚Ä¢ Windows Hello: ${capabilities.biometricAvailability.windowsHello}`)
    console.log(`  ‚Ä¢ Android Fingerprint: ${capabilities.biometricAvailability.androidFingerprint}`)
    console.log(`  ‚Ä¢ Android Face: ${capabilities.biometricAvailability.androidFace}`)
    
    console.log('\nüì± Device Information:')
    console.log(`  ‚Ä¢ Platform: ${capabilities.deviceInfo.platform}`)
    console.log(`  ‚Ä¢ iOS: ${capabilities.deviceInfo.isIOS}`)
    console.log(`  ‚Ä¢ Android: ${capabilities.deviceInfo.isAndroid}`)
    console.log(`  ‚Ä¢ macOS: ${capabilities.deviceInfo.isMacOS}`)
    console.log(`  ‚Ä¢ Windows: ${capabilities.deviceInfo.isWindows}`)
    console.log(`  ‚Ä¢ Mobile: ${capabilities.deviceInfo.isMobile}`)
    console.log(`  ‚Ä¢ OS Version: ${capabilities.deviceInfo.osVersion || 'Unknown'}`)
    
    // Provide recommendations
    console.log('\nüí° Recommendations:')
    if (!capabilities.isSupported) {
      console.log('‚ùå WebAuthn is not supported on this browser/device')
      console.log('   ‚Üí Use a modern browser (Chrome 67+, Firefox 60+, Safari 14+)')
    } else if (!capabilities.isPlatformAuthenticatorAvailable) {
      console.log('‚ö†Ô∏è  Platform authenticator not available')
      console.log('   ‚Üí Enable biometric authentication in device settings')
      console.log('   ‚Üí Ensure you\'re using HTTPS')

      console.log('‚ö†Ô∏è  No biometric methods detected')
      console.log('   ‚Üí Check device biometric settings')
    } else {
      console.log('‚úÖ Ready for biometric authentication!')
      console.log(`   ‚Üí Available methods: ${capabilities.biometricTypes.join(', ')}`)
    }
    
    return capabilities
    
  } catch (error) {
    console.error('‚ùå Error testing biometric capabilities:', error)

  }
}
 
export async function testBiometricRegistration(userId: string = 'test-user', userName: string = 'Test User') {
  try {
    console.log('\nüîê Testing biometric registration...')
    
    const result = await WebAuthnService.register({ userId, userName, userDisplayName: userName })
    
    // Result is direct data, not wrapped in success object
    
    console.log('‚úÖ Registration successful!')
    console.log(`   ‚Ä¢ Credential ID: ${result.id?.substring(0, 20)}...`)
    console.log(`   ‚Ä¢ Authenticator: ${result.authenticatorAttachment || 'unknown'}`)
    
    return result
    
  } catch (error: any) {
    console.error('‚ùå Registration failed:', error.message || error)
    if (error.code) {
      console.error(`   ‚Ä¢ Error Code: ${error.code}`)
      console.error(`   ‚Ä¢ Suggested Action: ${error.suggestedAction}`)
    }
    throw error
  }
}

// Convenience function to run all tests
export async function runBiometricTests() {
  try {
    // Test capabilities
    const capabilities = await testBiometricCapabilities()
    
    // Only proceed with registration/auth tests if biometrics are supported
    if (capabilities.isPlatformAuthenticatorAvailable && capabilities.biometricTypes.length > 0) {
      // Test registration
      await testBiometricRegistration()
      
      // Test authentication
      await testBiometricAuthentication()
      
      console.log('\nüéâ All biometric tests completed successfully!')
    } else {
      console.log('\n‚è≠Ô∏è  Skipping registration/authentication tests due to lack of biometric support')
    }
    
  } catch (error) {
    console.error('\nüí• Biometric test suite failed:', error)
  }
}
